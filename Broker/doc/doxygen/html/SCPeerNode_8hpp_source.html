00132 struct SCpeercomp {
00133   bool operator() (const SCPeerNodePtr& lhs, const SCPeerNodePtr& rhs) const
00134   /* Returns true if
00135    * 1) lhs has a greater NodeID
00136    *   OR
00137    * 2) lhs and rhs has equal node id, but lhs has a greater address
00138    * -This can occur after PeerNode is initialized, but has initial NodeID of 0
00139    * -This will update upon first received packet
00140    */
00141   {
00142       return (lhs->m_NodeID > rhs->m_NodeID) ||
00143           ((lhs->m_NodeID == rhs->m_NodeID) &&
00144                   !(lhs->m_endpoint.address() < rhs->m_endpoint.address())
00145                   && (lhs->m_endpoint.address() != rhs->m_endpoint.address()));
00146   }
00147 };
